28#if defined(_MSC_VER)
29//
30// Direct Show のビデオデバイスの一覧を作る
31//
32// https://docs.microsoft.com/en-us/windows/win32/directshow/selecting-a-capture-device
33// https://www.geekpage.jp/programming/directshow/list-capture-device.php
34// http://www.antillia.com/sol9.2.0/4.25.html
35//
36#include <dshow.h>
37#pragma comment(lib, "strmiids")
38
39void getDirectShowList(std::vector<std::string>& list)
40{
41 // COM を開く
42 HRESULT hr{ CoInitialize(nullptr) };
43 if (FAILED(hr)) return;
44
45 // デバイスの列挙子を作成する
46 ICreateDevEnum* pDevEnum{ nullptr };
47 hr = CoCreateInstance(CLSID_SystemDeviceEnum, NULL, CLSCTX_INPROC_SERVER, IID_ICreateDevEnum,
48 reinterpret_cast<PVOID*>(&pDevEnum));
49
50 // 列挙子が作れなかったら戻る
51 if (FAILED(hr)) return;
52
53 // デバイスの列挙子の異名を作成する
54 IEnumMoniker* pEnumMoniker{ nullptr };
55 hr = pDevEnum->CreateClassEnumerator(CLSID_VideoInputDeviceCategory, &pEnumMoniker, 0);
56
57 // デバイスの列挙子はもういらないので開放する
58 pDevEnum->Release();
59
60 // 列挙子の異名が作れなかったら戻る
61 if (FAILED(hr)) return;
62
63 // 列挙子の異名が一つもなければ戻る
64 if (!pEnumMoniker) return;
65
66 // 列挙子の異名の取り出し先
67 IMoniker* pMoniker{ nullptr };
68
69 // 列挙子の異名を一つずつ取り出す
70 while (pEnumMoniker->Next(1, &pMoniker, nullptr) == S_OK)
71 {
72 // プロパティバッグの場所を取り出す
73 IPropertyBag* pPropertyBag;
74 hr = pMoniker->BindToStorage(0, 0, IID_IPropertyBag, reinterpret_cast<void**>(&pPropertyBag));
75
76 // プロパティバッグの場所が取り出せなかった次に行く
77 if (FAILED(hr))
78 {
79 pMoniker->Release();
80 continue;
81 }
82
83 // FriendlyName の格納場所
84 VARIANT friendlyName;
85 VariantInit(&friendlyName);
86
87 // FriendlyName を取得する
88 hr = pPropertyBag->Read(L"FriendlyName", &friendlyName, 0);
89
90 // Friendly Name を保存する
91 list.emplace_back(FAILED(hr) ? "Unknown" : TCharToUtf8(friendlyName.bstrVal));
92
93 // FriendlyName の格納場所を消去する
94 VariantClear(&friendlyName);
95
96 // プロパティバッグを解放する
97 pMoniker->Release();
98 pPropertyBag->Release();
99 }
100
101 // デバイスの列挙子の異名を開放する
102 pEnumMoniker->Release();
103
104 // COM を閉じる
105 CoUninitialize();
106}

109// Media Foundation のビデオデバイスの一覧を作る
110//
111// https://docs.microsoft.com/ja-jp/windows/win32/medfound/audio-video-capture-in-media-foundation
112//
113#include <Mfidl.h>
114#include <Mfapi.h>
115#include <Mferror.h>
116#pragma comment(lib, "mf.lib")
117#pragma comment(lib, "mfplat.lib")
118
119void getMediaFoundationList(std::vector<std::string>& list)
120{
121 // Create an attribute store to hold the search criteria.
122 IMFAttributes* pConfig{ NULL };
123 HRESULT hr{ MFCreateAttributes(&pConfig, 1) };
124
125 // Request video capture devices.
126 if (SUCCEEDED(hr))
127 {
128 hr = pConfig->SetGUID(
129 MF_DEVSOURCE_ATTRIBUTE_SOURCE_TYPE,
130 MF_DEVSOURCE_ATTRIBUTE_SOURCE_TYPE_VIDCAP_GUID
131 );
132 }
133
134 // Enumerate the devices,
135 IMFActivate** ppDevices{ NULL };
136 UINT32 count{ 0 };
137 if (SUCCEEDED(hr))
138 {
139 hr = MFEnumDeviceSources(pConfig, &ppDevices, &count);
140 }
141
142 for (DWORD i = 0; i < count; i++)
143 {
144 // Try to get the display name.
145 WCHAR* szFriendlyName{ NULL };
146 UINT32 cchName{ 0 };
147 HRESULT hr{
148 ppDevices[i]->GetAllocatedString(MF_DEVSOURCE_ATTRIBUTE_FRIENDLY_NAME,
149 &szFriendlyName, &cchName)
150 };
151
152 if (SUCCEEDED(hr))
153 {
154 list.emplace_back(TCharToUtf8(szFriendlyName));
155 }
156 CoTaskMemFree(szFriendlyName);
157 }
158
159 for (DWORD i = 0; i < count; i++)
160 {
161 ppDevices[i]->Release();
162 }
163 CoTaskMemFree(ppDevices);
164}

192Config::Config(const std::string& filename)
193 : title{ PROJECT_NAME }
194 , windowSize{ 1280, 720 }
195 , background{ 0.2f, 0.3f, 0.4f, 1.0f }
196 , settings{ "DICT_4X4_50" }
197 , menuFont{ "Mplus1-Regular.ttf" }
198 , menuFontSize{ 20.0f }
199{
200 // バックエンドごとのキャプチャデバイスの一覧を初期化する
201 for (auto& [api, name] : backendList)
202 {
203 // バックエンドごとに空のリストを追加する
204 deviceList.emplace(api, std::vector<std::string>());
205 }
206
207 // キャプチャデバイスの一覧を作る
208 getAnyList(deviceList.at(cv::CAP_ANY));
209#if defined(_MSC_VER)
210 getDirectShowList(deviceList.at(cv::CAP_DSHOW));
211 getMediaFoundationList(deviceList.at(cv::CAP_MSMF));
212#elif defined(__APPLE__)
213 getAvFoundationList(deviceList.at(cv::CAP_AVFOUNDATION));
214#endif
215
216 // 構成ファイルの保存場所を決定する
217#if defined(_DEBUG)
218 const auto path{ Utf8ToTChar(filename) };
219#else
220# if defined(_MSC_VER)
221 // 構成ファイルの保存先のパス
222 wchar_t appDataPath[MAX_PATH];
223
224 // AppData のバスを取得する
225 SHGetSpecialFolderPathW(NULL, appDataPath, CSIDL_APPDATA, 0);
226
227 // AppData のパスに構成ファイル名を連結する
228 const auto path{ CString(appDataPath) + TEXT("\\") + Utf8ToTChar(filename) };
229# else
230 // パスワードファイルのエントリを取得する
231 const passwd* pw{ getpwuid(getuid()) };
232
233 // ホームディレクトリのパスに構成ファイル名を連結する
234 const auto path{ std::string(pw->pw_dir) + "/." + filename };
235# endif
236#endif
237
238 // 構成ファイルを読み込む
239 if (!load(path))
240 {
241 // デフォルトの設定を追加する
242 if (!load(Utf8ToTChar(filename))) preferenceList.emplace_back();
243
244 // デフォルトの設定を入れた構成ファイルを作成する
245 save(path);
246 }
247}

269// 構成ファイルを読み込む
270//
271bool Config::load(const pathString& filename)
272{
273 // 構成ファイルの読み込み
274 std::ifstream json{ filename };
275 if (!json) return false;
276
277 // JSON の読み込み
278 picojson::value value;
279 json >> value;
280 json.close();
281
282 // 構成内容の取り出し
283 const auto& object{ value.get<picojson::object>() };
284
285 // オブジェクトが空だったらエラー
286 if (object.empty()) return false;
287
288 // ウィンドウのサイズ
289 getValue(object, "size", windowSize);
290
291 // ウィンドウの背景色
292 getValue(object, "background", background);
293
294 // メッシュのサンプル数
295 getValue(object, "samples", settings.samples);
296
297 // キャプチャデバイスの姿勢
298 getValue(object, "pose", settings.euler);
299
300 // 描画時の焦点距離
301 getValue(object, "focal", settings.focal);
302
303 // 描画時の焦点距離の範囲
304 getValue(object, "range", settings.focalRange);
305
306 // ArUco Marker の辞書名
307 getString(object, "dictionary", settings.dictionaryName);
308
309 // 初期表示画像
310 getString(object, "initial", initialImage);
311
312 // FFmpeg のリスト
313 getString(object, "ffmpeg", deviceList[cv::CAP_FFMPEG]);
314
315 // GStreamer のリスト
316 getString(object, "gstreamer", deviceList[cv::CAP_GSTREAMER]);
317
318 // キャプチャデバイスの構成を探す
319 const auto& camera{ object.find("camera") };
320
321 // キャプチャデバイスの構成の配列が見つからなければエラー
322 if (camera == object.end() || !camera->second.is<picojson::array>()) return false;
323
324 // 配列の個々の要素について
325 for (const auto& value : camera->second.get<picojson::array>())
326 {
327 // キャプチャデバイスの構成のオブジェクトを取り出す
328 const auto& preference{ value.get<picojson::object>() };
329
330 // キャプチャデバイスの構成をリストに追加
331 preferenceList.emplace_back(preference);
332 }
333
334 // キャプチャデバイスの構成が一つも読み取れなければエラー
335 return !preferenceList.empty();
336}

341bool Config::save(const pathString& filename) const
342{
343 // 設定値を保存する
344 std::ofstream config{ filename };
345 if (!config) return false;
346
347 // オブジェクト
348 picojson::object object;
349
350 // ウィンドウのサイズ
351 setValue(object, "size", windowSize);
352
353 // ウィンドウの背景色
354 setValue(object, "background", background);
355
356 // メッシュのサンプル数
357 setValue(object, "samples", settings.samples);
358
359 // キャプチャデバイスの姿勢
360 setValue(object, "pose", settings.euler);
361
362 // 描画時の焦点距離
363 setValue(object, "focal", settings.focal);
364
365 // 描画時の焦点距離の範囲
366 setValue(object, "range", settings.focalRange);
367
368 // ArUco Marker 辞書名
369 setString(object, "dictionary", settings.dictionaryName);
370
371 // 初期表示画像
372 setString(object, "initial", initialImage);
373
374 // FFmpeg のリスト
375 setString(object, "ffmpeg", deviceList.at(cv::CAP_FFMPEG));
376
377 // GStreamer のリスト
378 setString(object, "gstreamer", deviceList.at(cv::CAP_GSTREAMER));
379
380 // 配列
381 picojson::array array;
382
383 // 構成リストのすべて構成について
384 for (const auto& preference : preferenceList)
385 {
386 // 構成の要素
387 picojson::object camera;
388
389 // 構成を JSON オブジェクトに格納する
390 preference.setPreference(camera);
391
392 // 要素を picojson::array に追加する
393 array.emplace_back(picojson::value(camera));
394 }
395
396 // オブジェクトに追加する
397 object.emplace("camera", array);
398
399 // 構成をシリアライズして保存
400 picojson::value v{ object };
401 config << v.serialize(true);
402 config.close();
403
404 // 構成データの書き込み成功
405 return true;
406}

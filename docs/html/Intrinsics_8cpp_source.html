32void Intrinsics::setFov(float focal)
33{
34 // 撮像面の画素数の対角線長×２
35 const auto d{ hypotf(static_cast<float>(size[0]), static_cast<float>(size[1])) * 2.0f };
36
37 // 撮像面の画素数の対角線長に対するスクリーンの幅と高さの比の 1/2 (d は対角線長の２倍だから)
38 const auto w{ size[0] / d };
39 const auto h{ size[1] / d };
40
41 // 焦点距離が１のときの撮像面の対角線長
42 const auto t{ sensorSize / focal };
43
44 // 焦点距離が１のときの撮像面の対角線長をこの比で横と縦に振り分ける
45 const auto u{ t * w };
46 const auto v{ t * h };
47
48 // 画角を求め単位を度に換算して設定する
49 constexpr auto s{ 360.0f / 3.14159265359f };
50 setFov(atan(u) * s, atan(v) * s);
51}

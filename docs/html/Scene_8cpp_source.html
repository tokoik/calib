19 , vbo{ [] { GLuint buffer; glGenBuffers(1, &buffer); return buffer; }() }
20 , vertexCount{ (slices + 1) * (stacks + 1) * instances }
21 , ibo{ [] { GLuint buffer; glGenBuffers(1, &buffer); return buffer; }() }
22 , drawCount{ stacks * instances }
23 , indexCount{ (slices + 1) * 2 * drawCount }
24{
25 // 頂点配列オブジェクト
26 glBindVertexArray(vao);
27
28 // 頂点の番号のバッファオブジェクトを作成する
29 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ibo);
30 glBufferData(GL_ELEMENT_ARRAY_BUFFER, indexCount * sizeof(GLushort), nullptr, GL_STATIC_DRAW);
31
32 // 頂点の番号のバッファオブジェクトに番号を格納する
33 auto index{ static_cast<GLushort*>(glMapBuffer(GL_ELEMENT_ARRAY_BUFFER, GL_WRITE_ONLY)) };
34
35 // メッシュの１ライン分の頂点数
36 const auto vertex_count_of_a_line{ slices + 1 };
37
38 // メッシュの１ストリップ分の頂点数
39 const auto vertex_count_of_a_strip{ vertex_count_of_a_line * 2 };
40
41 // ストリップとラインのそれぞれの最初の頂点番号
42 int strip{ 0 }, line{ 0 };
43
44 // すべての面について
45 for (int k = 0; k < instances; ++k)
46 {
47 // すべてのストリップについて
48 for (int j = 0; j < stacks; ++j)
49 {
50 // そのストリップの最初の頂点の番号をバッファオブジェクトに保存する
51 indices.emplace_back(static_cast<GLushort*>(0) + strip);
52
53 // そのストリップの頂点数を保存する
54 count.emplace_back(vertex_count_of_a_strip);
55
56 // そのストリップの個々の頂点について
57 for (int i = 0; i < vertex_count_of_a_line; ++i)
58 {
59 // ストリップの下側のラインの頂点番号
60 *index++ = line + i;
61
62 // ストリップの上側のラインの頂点番号
63 *index++ = line + i + vertex_count_of_a_line;
64 }
65
66 // 次のストリップの最初の頂点番号
67 strip += vertex_count_of_a_strip;
68
69 // 次のラインの最初の頂点番号
70 line += vertex_count_of_a_line;
71 }
72
73 // 次の面の最初のラインの最初の頂点番号
74 line += vertex_count_of_a_line;
75 }
76
77 // 頂点の番号のバッファオブジェクトに番号を格納を終了する
78 glUnmapBuffer(GL_ELEMENT_ARRAY_BUFFER);
79
80 // 頂点の位置のバッファオブジェクト
81 glBindBuffer(GL_ARRAY_BUFFER, vbo);
82 glBufferData(GL_ARRAY_BUFFER, vertexCount * sizeof(Vertex), nullptr, GL_STATIC_DRAW);
83 glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(Vertex), offsetptr(Vertex, position));
84 glEnableVertexAttribArray(0);
85 glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, sizeof(Vertex), offsetptr(Vertex, normal));
86 glEnableVertexAttribArray(1);
87}

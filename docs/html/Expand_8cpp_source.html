37std::array<int, 2> Expand::setup(int samples, GLfloat aspect, const gg::GgMatrix& pose,
38 const std::array<GLfloat, 2>& fov, const std::array<GLfloat, 2>& center, GLfloat focal,
39 const std::array<GLfloat, 4>& border, int unit) const
40{
41 // プログラムオブジェクトの指定
42 glUseProgram(program);
43
44 // テクスチャユニットの指定
45 glUniform1i(imageLoc, unit);
46
47 // 境界色
48 glUniform4fv(borderLoc, 1, border.data());
49
50 // 投影像の画角（度）と中心位置
51 glUniform4f(circleLoc, fov[0], fov[1], center[0], center[1]);
52
53 // スクリーンのサイズと中心位置
54 // screen[0] = (right - left) / 2
55 // screen[1] = (top - bottom) / 2
56 // screen[2] = (right + left) / 2
57 // screen[3] = (top + bottom) / 2
58 const GLfloat screen[]{ aspect, 1.0f, 0.0f, 0.0f };
59 glUniform4fv(screenLoc, 1, screen);
60
61 // レンズの主点とスクリーンの距離
62 glUniform1f(focalLoc, focal);
63
64 // 背景に対する視線の回転行列
65 glUniformMatrix4fv(rotationLoc, 1, GL_FALSE, pose.get());
66
67 // メッシュの横の格子点数
68 // 標本点の数 (頂点数) samples = w * h とするとき、これに縦横比
69 // aspect = w / h をかければ aspect * samples = w * w となるから、
70 // w = sqrt(aspect * samples), h = samples / w で求められる
71 const auto w{ static_cast<int>(sqrt(aspect * samples)) };
72 const auto h{ samples / w };
73
74 // 格子間隔
75 // クリッピング空間全体を埋める四角形は [-1, 1] の範囲すなわち
76 // 縦横 2 の大きさだから、 それを縦横の (格子数 - 1) で割る
77 std::array<GLfloat, 2> gap{ 2.0f / (w - 1), 2.0f / (h - 1) };
78 glUniform2fv(gapLoc, 1, gap.data());
79
80 // 描画するメッシュの横と縦の格子点数を返す
81 return std::array<int, 2>{ w, h };
82}

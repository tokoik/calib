24void Menu::loadConfig()
25{

27 constexpr nfdfilteritem_t jsonFilter[]{ { "JSON", "json" } };
28
29 // ファイルダイアログから得るパス
30 nfdchar_t* filepath;

34 {

36 if (const_cast<Config&>(config).load(filepath))

43 settings = config.settings;
44 }
45 else

47 // 読み込めなかった
48 errorMessage = u8"構成ファイルが読み込めません";
49 }
50
51 // ファイルパスの取り出しに使ったメモリを開放する
52 NFD_FreePath(filepath);

118void Menu::loadMovie()
119{
120 // 動画ファイル名のフィルタ
121 constexpr nfdfilteritem_t movieFilter[]{ "Movies", "mp4,m4v,mpg,mov,avi,ogg,mkv" };
122
123 // ファイルダイアログから得るパス
124 nfdchar_t* filepath;
125
126 // ファイルダイアログを開く
127 if (NFD_OpenDialog(&filepath, movieFilter, 1, NULL) == NFD_OKAY)
128 {
129 // 入力特性をファイルに切り替えて
130 backend = cv::CAP_FFMPEG;
131
132 // ファイルのリストを取り出し
133 auto& fileList{ config.deviceList.at(backend) };
134
135 // ファイルのリストの各ファイルについて
136 for (deviceNumber = 0; deviceNumber < static_cast<int>(fileList.size()); ++deviceNumber)
137 {
138 // 選択したファイルと同じものがあればそれを選択する
139 if (fileList[deviceNumber] == filepath) break;
140 }
141
142 // 選択したファイルがファイルのリストの中になければ
143 if (deviceNumber == static_cast<int>(fileList.size()))
144 {
145 // その先頭にファイルパスを挿入して
146 fileList.insert(fileList.begin(), filepath);
147
148 // そのエントリを選択する
149 deviceNumber = 0;
150 }
151
152 // ファイルパスの取り出しに使ったメモリを開放する
153 NFD_FreePath(filepath);
154 }

212// キャプチャを開始する
213//
214void Menu::startCapture()
215{
216 // キャプチャ中ならキャプチャを止めて
217 stopCapture();
218
219 // キャプチャデバイスのリストが空だったら戻る
220 if (config.deviceList.at(backend).empty()) return;
221
222 // 新しいキャプチャデバイスを作成したら
223 auto camCv{ std::make_unique<CamCv>() };
224
225 // キャプチャデバイスを開く
226 switch (backend)
227 {
228 case cv::CAP_FFMPEG:
229 // バックエンドが FFmpeg ならキャプチャデバイス名の文字列でデバイスを開く
230 if (!camCv->open(config.getDeviceName(backend, deviceNumber))) return;
231 break;
232 case cv::CAP_GSTREAMER:
233 // バックエンドが GStreamer ならキャプチャデバイス名の文字列でデバイスを開く
234 if (!camCv->open(config.getDeviceName(backend, deviceNumber),
235 0, 0, 0.0, "", backend)) return;
236 break;
237 default:
238 // キャプチャデバイス番号でデバイスを開く
239 if (!camCv->open(deviceNumber,
240 intrinsics.resolution[0],
241 intrinsics.resolution[1],
242 intrinsics.fps,
243 codecNumber == 0 ? "" : config.codecList[codecNumber],
244 backend)) return;
245 break;
246 }
247
248 // 開いたキャプチャデバイス固有のパラメータを構成データに保存して
249 intrinsics.setResolution(camCv->getWidth(), camCv->getHeight());
250 intrinsics.setFps(camCv->getFps());
251
252 // フレームの格納先のフレームバッファオブジェクトを作り直してから
253 framebuffer.create(camCv->getWidth(), camCv->getHeight(), camCv->getChannels(), nullptr);
254
255 // キャプチャを開始する
256 camCv->start();
257
258 // このキャプチャデバイスを使うことにする
259 camera = std::move(camCv);
260}

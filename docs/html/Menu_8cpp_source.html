35void Menu::openDevice()

38 if (backend == cv::CAP_GSTREAMER)
39 {
40 // ファイルのリストを取り出し

56 // 開けなかった
57 errorMessage = u8"キャプチャデバイスが開けません";

96 nfdchar_t* filepath;

98 // ファイルダイアログを開く
99 if (NFD_OpenDialog(&filepath, movieFilter, 1, NULL) == NFD_OKAY)
100 {
101 // 入力特性をファイルに切り替えて
102 backend = cv::CAP_FFMPEG;
103
104 // ファイルのリストを取り出し
105 auto& fileList{ config.deviceList.at(backend) };
106 const auto fileListLength{ static_cast<int>(fileList.size()) };
107
108 // ファイルのリストの各ファイルについて
109 for (deviceNumber = 0; deviceNumber < fileListLength; ++deviceNumber)
110 {
111 // 選択したファイルと同じものがあればそれを選択する
112 if (fileList[deviceNumber] == filepath) break;
113 }
114
115 // 選択したファイルがファイルのリストの中になければ
116 if (deviceNumber == fileListLength)
117 {
118 // その先頭にファイルパスを挿入して
119 fileList.insert(fileList.begin(), filepath);
120
121 // そのエントリを選択する
122 deviceNumber = 0;
123 }
124
125 // ダイアログで指定した動画ファイルが開けたら
126 if (capture.openMovie(filepath, backend))
127 {
128 // 構成データの解像度と画角を開いた画像に合わせる
129 setSize(capture.getSize());
130 }
131 else
132 {
133 // 開けなかった
134 errorMessage = u8"動画ファイルが開けません";
135 }
136
137 // ファイルパスの取り出しに使ったメモリを開放する
138 NFD_FreePath(filepath);

226void Menu::saveParameters() const
227{
228 // キャリブレーションが完了していなければ戻る
229 if (!calibration.finished()) return;
230
231 // 現在時刻の取得
232 const auto now{ std::chrono::system_clock::to_time_t(std::chrono::system_clock::now()) };
233 const std::tm* const localTime{ std::localtime(&now) };
234
235 // 時刻を文字列に変換
236 const auto timeString{ std::put_time(localTime, "cal_%Y%m%d_%H%M%S.json") };
237 const auto pathString{ static_cast<std::ostringstream&&>(std::ostringstream() << timeString).str() };
238
239 // ファイルダイアログから得るパス
240 nfdchar_t* filepath;
241
242 // ファイルダイアログを開く
243 if (NFD_SaveDialog(&filepath, jsonFilter, 1, NULL, pathString.c_str()) == NFD_OKAY)
244 {
245 // 現在のキャリブレーションパラメータを構成ファイルに保存する
246 if (!calibration.saveParameters(filepath))
247 {
248 // 保存できなかった
249 errorMessage = u8"較正ファイルが保存できません";
250 }
251
252 // ファイルパスの取り出しに使ったメモリを開放する
253 NFD_FreePath(filepath);

257//
258// 較正用の画像ファイルを取得する (複数選択)
259//
260void Menu::recordFileCorners() const
261{
262 // ファイルダイアログから得るパス
263 const nfdpathset_t* outPaths;
264
265 // ファイルダイアログを開く
266 if (NFD_OpenDialogMultiple(&outPaths, imageFilter, 1, NULL) == NFD_OKAY)
267 {
268 // ファイルパスの一覧を得る
269 nfdpathsetenum_t enumerator;
270 NFD_PathSet_GetEnum(outPaths, &enumerator);
271
272 // ファイルパスの一覧からファイルパスを一つずつ取り出して
273 for (nfdchar_t* path = NULL; NFD_PathSet_EnumNext(&enumerator, &path) && path;)
274 {
275 // 画像ファイルを読み出す
276 auto image{ CamImage::load(path) };
277
278 // 読み出したファイルに中身が入っていたら
279 if (!image.empty())
280 {
281 // 解析用のバッファを作って
282 Buffer buffer{ image.cols, image.rows, image.channels() };
283
284 // バッファにデータを書き込んで
285 buffer.setData(image.cols * image.rows * image.channels(), image.data);
286
287 // ボードを検出して
288 calibration.detectBoard(buffer);
289
290 // コーナーを記録する
291 calibration.recordCorners();
292 }
293
294 // ファイルパスの取り出しに使ったメモリを開放する
295 NFD_PathSet_FreePath(path);
296 }
297
298 // ファイルパスの一覧に使ったメモリを開放する
299 NFD_PathSet_FreeEnum(&enumerator);
300
301 // フォルダのパスに使ったメモリを開放する
302 NFD_PathSet_Free(outPaths);
303 }
304}

768 calibration.recordCorners();
769 }

772 ImGui::Render();

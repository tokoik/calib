88 void capture()
89 {
90 // 再生開始時刻
91 auto startTime{ glfwGetTime() };
92
93 // スレッドが実行可の間
94 while (running)
95 {
96 // フレームを取り出せたら true
97 auto status{ (total <= 0.0 || camera.get(cv::CAP_PROP_POS_FRAMES) < out) && camera.grab() };
98
99 // ムービーファイルでないかムービーファイルの終端でなければ次のフレームを取り出して
100 if (status && camera.retrieve(frame))
101 {
102 // ピクセルバッファオブジェクトをロックしてから
103 std::lock_guard lock{ mtx };
104
105 // 転送用の一時メモリにデータを格納したら
106 copyFrame();
107
108 // 新しいフレームがキャプチャされたことを通知する
109 captured = true;
110 }
111
112 // 遅延時間
113 auto deferred{ 0.0 };
114
115 // ムービーファイルから入力しているとき
116 if (total > 0.0)
117 {
118 // ムービーファイルの終端に到達していたら
119 if (!status)
120 {
121 // インポイントまで巻き戻す
122 camera.set(cv::CAP_PROP_POS_FRAMES, in);
123
124 // 経過時間を戻す
125 elapsedTime = 0.0;
126
127 // 開始時間を更新する
128 startTime = glfwGetTime();
129 }
130 else
131 {
132 // 現在のフレームのインポイントからの経過時間
133 const auto pos{ camera.get(cv::CAP_PROP_POS_MSEC) - in };
134
135 // 再生位置の次のフレームの時刻に対する経過時間
136 const auto now{ (elapsedTime + glfwGetTime() - startTime) * 1000.0 + interval };
137
138 // 遅延時間はフレームの経過時間と現在の経過時間の差
139 deferred = pos - now;
140 }
141 }
142
143 // 遅延時間あれば
144 if (deferred > 0.0)
145 {
146 // 待つ
147 std::this_thread::sleep_for(std::chrono::milliseconds(static_cast<int>(deferred)));
148 }
149 }
150
151 // 再生停止までの経過時間を積算する
152 elapsedTime += glfwGetTime() - startTime;
153 }

100 void capture()
101 {
102 // スレッドが実行可の間
103 while (run)
104 {
105 // フレームを取り出せたら true
106 auto status{ (frames <= 0.0 || camera.get(cv::CAP_PROP_POS_FRAMES) < out) && camera.grab() };
107
108 // ムービーファイルでないかムービーファイルの終端でなければ次のフレームを取り出して
109 if (status && camera.retrieve(frame))
110 {
111 // ピクセルバッファオブジェクトをロックしてから
112 std::lock_guard lock{ mtx };
113
114 // 転送用の一時メモリにデータを格納したら
115 copyFrame(frame);
116
117 // 新しいフレームがキャプチャされたことを通知する
118 captured = true;
119 }
120
121 // 現在時刻
122 const auto now{ glfwGetTime() * 1000.0 };
123
124 // 遅延時間
125 auto deferred{ startTime + interval - now };
126
127 // ムービーファイルから入力しているとき
128 if (frames > 0.0)
129 {
130 // ムービーファイルの終端に到達していたら
131 if (!status)
132 {
133 // インポイントまで巻き戻す
134 camera.set(cv::CAP_PROP_POS_FRAMES, in);
135
136 // 巻き戻した後のフレーム時刻
137 const auto pos{ in * interval };
138
139 // インポイントを現在時刻としたときの開始時刻を求める
140 startTime = now - pos;
141
142 // インポイントでは遅延させない
143 deferred = 0.0;
144 }
145 else
146 {
147 // 現在のフレーム時刻
148 const auto pos{ camera.get(cv::CAP_PROP_POS_MSEC) };
149
150 // 開始時刻から見た現在のフレームの遅延を求める
151 deferred += pos;
152 }
153 }
154 else
155 {
156 // ムービーファイルでなければ現在の時刻を開始時刻にする
157 startTime = now;
158 }
159
160 // 遅延時間あれば
161 if (deferred > 0.0)
162 {
163 // 待つ
164 std::this_thread::sleep_for(std::chrono::milliseconds(static_cast<int>(deferred)));
165 }
166 }

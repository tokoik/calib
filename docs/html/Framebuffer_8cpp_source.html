113// フレームバッファオブジェクトを作成する
114//
115void Framebuffer::create(GLsizei width, GLsizei height, int channels)
116{
117 // 既存のテクスチャを破棄して新しいテクスチャを作成する
118 Texture::create(width, height, channels);
119
120 // まだメッシュの頂点配列オブジェクトが作られていなければ作る
121 if (!mesh) mesh = std::make_shared<Mesh>();
122
123 // 指定したサイズがフレームバッファオブジェクトのサイズと同じなら何もしない
124 if (width == framebufferSize[0] && height == framebufferSize[1]
125 && channels == framebufferChannels) return;
126
127 // フレームバッファオブジェクトのサイズとチャンネル数を記録する
128 framebufferSize = std::array<int, 2>{ width, height };
129 framebufferChannels = channels;
130
131 // 以前のフレームバッファオブジェクトを削除する
132 glDeleteFramebuffers(1, &framebufferName);
133
134 // フレームバッファオブジェクトを作成する
135 glGenFramebuffers(1, &framebufferName);
136 glBindFramebuffer(GL_FRAMEBUFFER, framebufferName);
137 glFramebufferTexture2D(GL_FRAMEBUFFER, attachment, GL_TEXTURE_2D, getTextureName(), 0);
138 glDrawBuffers(1, &attachment);
139 glReadBuffer(attachment);
140 glBindFramebuffer(GL_FRAMEBUFFER, 0);
141}

218//
219// フレームバッファオブジェクトの表示
220//
221void Framebuffer::show(GLsizei width, GLsizei height) const
222{
223 // フレームバッファオブジェクトの縦横比
224 const auto f{ static_cast<float>(framebufferSize[0] * height) };
225
226 // ウィンドウの表示領域の縦横比
227 const auto d{ static_cast<float>(framebufferSize[1] * width) };
228
229 // 実際に描画する領域
230 GLint dx0, dy0, dx1, dy1;
231
232 // 表示領域の右上端の位置を求める
233 --width;
234 --height;
235
236 // フレームバッファオブジェクトの縦横比が大きかったら
237 if (f > d)
238 {
239 // ウィンドウの表示領域の高さを求める
240 const auto h{ static_cast<GLint>(d / getWidth() + 0.5f) };
241
242 // 表示が横長なので表示領域の横幅いっぱいに表示する
243 dx0 = 0;
244 dx1 = width;
245
246 // 高さは縦横比を維持して描画する領域の中央に描く
247 dy0 = (height - h) / 2;
248 dy1 = dy0 + h;
249 }
250 else
251 {
252 // ウィンドウの表示領域の幅を求める
253 const auto w{ static_cast<GLint>(f / getHeight() + 0.5f) };
254
255 // 表示が縦長なので表示領域の高さいっぱいに表示する
256 dy0 = 0;
257 dy1 = height;
258
259 // 横幅は縦横比を維持して描画する領域の中央に描く
260 dx0 = (width - w) / 2;
261 dx1 = dx0 + w;
262 }
263
264 // フレームバッファオブジェクトを読み込み元にする
265 glBindFramebuffer(GL_READ_FRAMEBUFFER, framebufferName);
266 glReadBuffer(attachment);
267
268 // 現在のフレームバッファを背景色で塗りつぶす
269 glClear(GL_COLOR_BUFFER_BIT);
270
271 // フレームバッファオブジェクトの内容を通常のフレームバッファに書き込む
272 glBlitFramebuffer(0, 0, getWidth() - 1, getHeight() - 1,
273 dx0, dy1, dx1, dy0, GL_COLOR_BUFFER_BIT, GL_NEAREST);
274
275 // 読み込み元を通常のフレームバッファに戻す
276 glBindFramebuffer(GL_READ_FRAMEBUFFER, 0);
277 glReadBuffer(GL_BACK);
278}

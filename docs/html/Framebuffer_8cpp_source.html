130{
131 // 既存のテクスチャを破棄して新しいテクスチャを作成する
132 Texture::create(width, height, channels);
133
134 // 指定したサイズがフレームバッファオブジェクトのサイズと同じなら何もしない
135 if (width == framebufferSize[0] && height == framebufferSize[1]
136 && channels == framebufferChannels) return;
137
138 // フレームバッファオブジェクトのサイズとチャンネル数を記録する
139 framebufferSize = std::array<int, 2>{ width, height };
140 framebufferChannels = channels;

142 // 以前のフレームバッファオブジェクトを削除する
143 glDeleteFramebuffers(1, &framebufferName);
144

148 glFramebufferTexture(GL_FRAMEBUFFER, attachment, getTextureName(), 0);

228//
229// フレームバッファオブジェクトの表示
230//
231void Framebuffer::draw(GLsizei width, GLsizei height) const
232{
233 // フレームバッファオブジェクトの縦横比
234 const auto f{ static_cast<float>(getWidth() * height) };
235
236 // ウィンドウの表示領域の縦横比
237 const auto d{ static_cast<float>(getHeight() * width) };
238
239 // 描画する領域
240 GLint dx0, dy0, dx1, dy1;
241
242 // 表示領域の右上端の位置を求める
243 --width;
244 --height;
245
246 // フレームバッファオブジェクトの縦横比が大きかったら
247 if (f > d)
248 {
249 // ディスプレイ上の描画する領域の高さを求める
250 const auto h{ static_cast<GLint>(d / getWidth() + 0.5f) };
251
252 // 表示が横長なので描画する領域の横幅いっぱいに表示する
253 dx0 = 0;
254 dx1 = width;
255
256 // 高さは縦横比を維持して描画する領域の中央に描く
257 dy0 = (height - h) / 2;
258 dy1 = dy0 + h;
259 }
260 else
261 {
262 // ディスプレイ上の描画する領域の幅を求める
263 const auto w{ static_cast<GLint>(f / getHeight() + 0.5f) };
264
265 // 表示が縦長なので描画する領域の高さいっぱいに表示する
266 dy0 = 0;
267 dy1 = height;
268
269 // 横幅は縦横比を維持して描画する領域の中央に描く
270 dx0 = (width - w) / 2;
271 dx1 = dx0 + w;
272 }
273
274 // フレームバッファオブジェクトを読み込み元にする
275 glBindFramebuffer(GL_READ_FRAMEBUFFER, framebufferName);
276 glReadBuffer(attachment);
277
278 // 現在のフレームバッファを消去する
279 glClear(GL_COLOR_BUFFER_BIT);
280
281 // フレームバッファオブジェクトの内容を通常のフレームバッファに書き込む
282 glBlitFramebuffer(0, 0, getWidth() - 1, getHeight() - 1,
283 dx0, dy1, dx1, dy0, GL_COLOR_BUFFER_BIT, GL_NEAREST);
284
285 // 読み込み元を通常のフレームバッファに戻す
286 glBindFramebuffer(GL_READ_FRAMEBUFFER, 0);
287 glReadBuffer(GL_BACK);
288}

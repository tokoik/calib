36void Buffer::copyBuffer(const Buffer& buffer) noexcept
37{
38#if defined(USE_PIXEL_BUFFER_OBJECT)
39 // コピー元のピクセルバッファオブジェクト
40 glBindBuffer(GL_COPY_READ_BUFFER, buffer.bufferName);
41
42 // コピー元のピクセルバッファオブジェクトのサイズを調べる
43 GLint readSize;
44 glGetBufferParameteriv(GL_COPY_READ_BUFFER, GL_BUFFER_SIZE, &readSize);
45
46 // コピー先のピクセルバッファオブジェクト
47 glBindBuffer(GL_COPY_WRITE_BUFFER, bufferName);
48
49 // コピー先のピクセルバッファオブジェクトのサイズを調べる
50 GLint writeSize;
51 glGetBufferParameteriv(GL_COPY_WRITE_BUFFER, GL_BUFFER_SIZE, &writeSize);
52
53 // バッファサイズが一致しているか確認する
54 assert(readSize == writeSize);
55
56 // バッファオブジェクトをコピーする
57 glCopyBufferSubData(GL_COPY_READ_BUFFER, GL_COPY_WRITE_BUFFER,
58 0, 0, std::min(readSize, writeSize));
59#else
60 // フレームのデータをコピーする
61 memcpy(bufferName.data(), buffer.bufferName.data(), bufferName.size());
62#endif
63}

65//
66// バッファを作成する
67//
68void Buffer::create(GLsizei width, GLsizei height, int channels)
69{
70 // 指定したサイズが保持しているフレームのサイズと同じなら何もしない
71 if (width == bufferSize[0] && height == bufferSize[1]
72 && channels == bufferChannels) return;
73
74 // フレームのサイズとチャンネル数を記録する
75 bufferSize = std::array<int, 2>{ width, height };
76 bufferChannels = channels;
77
78 // フレームの保存に必要なメモリ量を求める
79 const auto size{ width * height * channels };
80
81#if defined(USE_PIXEL_BUFFER_OBJECT)
82 // 以前のピクセルバッファオブジェクトを破棄する
83 glDeleteBuffers(1, &bufferName);
84
85 // 新しいピクセルバッファオブジェクトを作成する
86 glGenBuffers(1, &bufferName);
87
88 // ピクセルバッファオブジェクトのメモリを確保してデータを転送する
89 glBindBuffer(GL_PIXEL_PACK_BUFFER, bufferName);
90 glBufferData(GL_PIXEL_PACK_BUFFER, size, nullptr, GL_DYNAMIC_COPY);
91 glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
92#else
93 // メモリを確保する
94 bufferName.resize(size);
95#endif
96}

183void Buffer::getData(GLsizei size, GLvoid* data) const
184{
185#if defined(USE_PIXEL_BUFFER_OBJECT)
186 // コピー元のピクセルバッファオブジェクトを結合する
187 glBindBuffer(GL_PIXEL_UNPACK_BUFFER, bufferName);
188
189 // コピー元のピクセルバッファオブジェクトのサイズを調べる
190 GLint readSize;
191 glGetBufferParameteriv(GL_PIXEL_UNPACK_BUFFER, GL_BUFFER_SIZE, &readSize);
192
193 // コピー元とコピー先の小さい方のサイズをコピーする
194 glGetBufferSubData(GL_PIXEL_UNPACK_BUFFER, 0, std::min(size, readSize), data);
195
196 // コピー元のピクセルバッファオブジェクトの結合を解除する
197 glBindBuffer(GL_PIXEL_UNPACK_BUFFER, 0);
198#else
199 // コピー元とコピー先の小さい方のサイズ
200 const auto bufferSize{ static_cast<GLsizei>(bufferName.size()) };
201 if (size > bufferSize) size = bufferSize;
202
203 // コピー元のメモリから引数に指定したコピー先の配列にコピーする
204 memcpy(data, bufferName.data(), size);
205#endif

20// 新しいテクスチャを作成する
21//
22GLuint Texture::createTexture(GLsizei width, GLsizei height, int channels, const GLvoid* pixels)
23{
24 // テクスチャのサイズとフォーマットを記録する
25 size = cv::Size2i{ width, height };
26 format = channelsToFormat(channels);
27
28 // テクスチャを作成する
29 glGenTextures(1, &name);
30 glBindTexture(GL_TEXTURE_2D, name);
31 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, format, GL_UNSIGNED_BYTE, pixels);
32 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
33 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
34 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
35 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
36 glBindTexture(GL_TEXTURE_2D, 0);
37
38#if defined(USE_PIXEL_BUFFER_OBJECT)
39 // フレームバッファの読み出しに使うピクセルバッファオブジェクトを作成する
40 glGenBuffers(1, &buffer);
41 glBindBuffer(GL_PIXEL_PACK_BUFFER, buffer);
42 glBufferData(GL_PIXEL_PACK_BUFFER, width * height * channels, nullptr, GL_DYNAMIC_COPY);
43 glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
44#else
45 buffer.resize(static_cast<std::vector<GLubyte>::size_type>(width) * height * channels);
46#endif
47
48 // テクスチャ名を返す
49 return getName();
50}

88template<typename MediaType>
89bool Texture::loadMedia(const std::string& filename)
90{
91 // キャプチャデバイスを作成して
92 auto media{ std::make_unique<MediaType>() };
93
94 // キャプチャデバイスが使えなかったら戻る
95 if (!media->open(filename)) return false;
96
97 // 画像のサイズとフォーマットを取り出して
98 const auto width{ media->getWidth() };
99 const auto height{ media->getHeight() };
100 const auto channels{ media->getChannels() };
101
102 // 描画するフレームを格納するテクスチャを作成したら
103 create(width, height, channels, nullptr);
104
105 // そこに読み込んだ画像をピクセルバッファオブジェクトに転送する
106 media->transmit(buffer);
107
108 // 読み込み成功
109 return true;
110}

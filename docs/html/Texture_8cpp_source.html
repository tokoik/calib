47// テクスチャを作成する
48//
49void Texture::create(GLsizei width, GLsizei height, int channels)
50{
51 // 既存のバッファを破棄して新しいバッファを作成する
52 Buffer::create(width, height, channels);
53
54 // まだメッシュの頂点配列オブジェクトが作られていなければ作る
55 if (!mesh) mesh = std::make_shared<Mesh>();
56
57 // 指定したサイズが保持しているテクスチャのサイズと同じなら何もしない
58 if (width == textureSize[0] && height == textureSize[1]
59 && channels == textureChannels) return;
60
61 // テクスチャのサイズとチャンネル数を記録する
62 textureSize = std::array<int, 2>{ width, height };
63 textureChannels = channels;
64
65 // 以前のテクスチャを削除する
66 glDeleteTextures(1, &textureName);
67
68 // テクスチャを作成する
69 glGenTextures(1, &textureName);
70
71 // テクスチャのメモリを確保してデータを転送する
72 glBindTexture(GL_TEXTURE_2D, textureName);
73 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0,
74 channelsToFormat(channels), GL_UNSIGNED_BYTE, nullptr);
75 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
76 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
77 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
78 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
79 glBindTexture(GL_TEXTURE_2D, 0);
80}

150//
151// このテクスチャをマッピングしてメッシュを描画する
152//
153void Texture::draw(GLsizei width, GLsizei height, int unit) const
154{
155 // テクスチャの縦横比
156 const auto t{ static_cast<float>(textureSize[0] * height)};
157
158 // 表示領域の縦横比
159 const auto d{ static_cast<float>(textureSize[1] * width)};
160
161 // テクスチャのスケール
162 const std::array<GLfloat, 2> scale{ t > d // テクスチャの縦横比の方が
163 ? std::array<GLfloat, 2>{ 1.0f, t / d } // 大きければ横方向いっぱいに描く
164 : std::array<GLfloat, 2>{ d / t, 1.0f } // それ以外は縦方向いっぱいに描く
165 };
166
167 // このオブジェクトのテクスチャを指定する
168 bindTexture(unit);
169
170 // メッシュを描画する
171 mesh->draw(scale, unit);
172
173 // テクスチャの指定を解除する
174 unbindTexture();
175}

178// テクスチャからピクセルバッファオブジェクトにデータをコピーする
179//
180void Texture::readPixels(
181#if defined(USE_PIXEL_BUFFER_OBJECT)
182 GLuint buffer) const
183#else
184std::vector<GLubyte>& buffer)
185#endif
186{
187 // 読み出し元のテクスチャを結合する
188 glBindTexture(GL_TEXTURE_2D, textureName);
189
190#if defined(USE_PIXEL_BUFFER_OBJECT)
191 // 書き込み先のピクセルバッファオブジェクトを指定する
192 glBindBuffer(GL_PIXEL_PACK_BUFFER, buffer);
193
194# if defined(GL_GLES_PROTOTYPES)
195 // 現在使っているフレームバッファオブジェクトを調べる
196 GLuint currentFbo;
197 glGetIntegerv(GL_FRAMEBUFFER_BINDING, reinterpret_cast<GLint*>(&currentFbo));
198
199 // テクスチャをカラーバッファに使ってフレームバッファオブジェクトを作る
200 GLuint fbo;
201 glGenFramebuffers(1, &fbo);
202 glBindFramebuffer(GL_FRAMEBUFFER, fbo);
203 static const GLenum attachment{ GL_COLOR_ATTACHMENT0 };
204 glFramebufferTexture2D(GL_FRAMEBUFFER, attachment, GL_TEXTURE_2D, textureName, 0);
205 glDrawBuffers(1, &attachment);
206 glReadBuffer(attachment);
207
208 // フレームバッファオブジェクトからピクセルバッファオブジェクトに読み出す
209 glReadPixels(0, 0, textureSize[0], textureSize[1], channelsToFormat(textureChannels),
210 GL_UNSIGNED_BYTE, 0);
211
212 // 元のフレームバッファオブジェクトに戻す
213 glBindFramebuffer(GL_FRAMEBUFFER, currentFbo);
214
215 // 作ったフレームバッファオブジェクトは削除する
216 glDeleteFramebuffers(1, &fbo);
217# else
218 // テクスチャの内容をピクセルバッファオブジェクトに書き込む
219 glGetTexImage(GL_TEXTURE_2D, 0, getFormat(), GL_UNSIGNED_BYTE, 0);
220# endif
221
222 // 書き込み先のピクセルバッファオブジェクトの結合を解除する
223 glBindBuffer(GL_PIXEL_PACK_BUFFER, 0);
224
225#else
226
227 // テクスチャの内容をピクセルバッファオブジェクトに書き込む
228 glGetTexImage(GL_TEXTURE_2D, 0, getFormat(), GL_UNSIGNED_BYTE, buffer.data());
229
230#endif
231
232 // 読み出し元のテクスチャの結合を解除する
233 glBindTexture(GL_TEXTURE_2D, 0);
234}

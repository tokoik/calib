116 void copyFrame(const cv::Mat& frame)
117 {
118 // 取り出したフレームのチャネル数を保存する
119 channels = frame.channels();
120
121 // 転送用の一時メモリのサイズを求める
122 const auto size{ resolution[0] * resolution[1] * channels };
123
124 // フレームの大きさを求める
125 const auto length{ frame.cols * frame.rows * channels };
126
127 // コピーするサイズを決める
128 const auto count{ std::min(size, length) };
129
130 // 転送用に必要なメモリサイズが以前と違ったらメモリを確保しなおす
131 if (pixels.size() != static_cast<decltype(pixels.size())>(count)) pixels.resize(count);
132
133 // R と B を入れ替えてコピーする
134 if (channels > 0)
135 {
136 if (channels < 3)
137 std::copy(frame.data, frame.data + count, pixels.data());
138 else
139 {
140 for (int i = 0; i < count; i += channels)
141 {
142 pixels.data()[i + 0] = frame.data[i + 2];
143 pixels.data()[i + 1] = frame.data[i + 1];
144 pixels.data()[i + 2] = frame.data[i + 0];
145 if (channels >= 3) pixels.data()[i + 3] = 1;
146 }
147 }
